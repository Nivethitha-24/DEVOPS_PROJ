pipeline {
    agent any

    stages {
        stage('Clone Repository') {
            steps {
                git credentialsId: 'github-token', url: 'https://github.com/YOUR_USERNAME/YOUR_REPO_NAME.git'
            }
        }

        stage('Deploy to Local Web Server') {
            steps {
                sh 'sudo cp -r * /var/www/html/'
            }
        }
    }
}
